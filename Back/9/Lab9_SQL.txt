CREATE TABLE burgers (
     id BIGINT GENERATED BY DEFAULT AS IDENTITY,
     name VARCHAR(50),
     price INTEGER,
     gram INTEGER,
     kcal INTEGER,
     protein INTEGER,
     PRIMARY KEY (id)
);
